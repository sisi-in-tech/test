HOW TO FROM ONE TABLE IMPORT DATA TO ANOTHER TABLE 
CREATE PROCEDURE P_DIM_SERIES 
AS 
BEGIN 

--SELECT * FROM DIM_SERIES

UPDATE D
SET  FOOTNOTE_CODES=A.footnote_codes, 
	 SERIES_NAME=A.series_title,
	 BEGIN_YEAR=A.begin_year,
	 END_YEAR= A.end_year,
	 DIM_GEOGRAPHIC_AREA_CODE=A.area_code,
	 DIM_ITEM_CODE=A.item_code,
	 BEGIN_DIM_TIME_PERIOD_CODE = A.begin_period, 
	 END_DIM_TIME_PERIOD_CODE = A.end_period
FROM ap_series A , 
     DIM_SERIES D 
WHERE A.[series_id        ]=D.DIM_SERIES_CODE

INSERT DIM_SERIES 
(DIM_SERIES_CODE, FOOTNOTE_CODES,SERIES_NAME,BEGIN_YEAR,END_YEAR, DIM_GEOGRAPHIC_AREA_CODE, DIM_ITEM_CODE, BEGIN_DIM_TIME_PERIOD_CODE, END_DIM_TIME_PERIOD_CODE )
SELECT DIM_SERIES_CODE = A.[series_id        ],
 FOOTNOTE_CODES=A.footnote_codes, 
 SERIES_NAME=A.series_title,
 BEGIN_YEAR=A.begin_year,
 END_YEAR= A.end_year,
 DIM_GEOGRAPHIC_AREA_CODE=A.area_code,
 DIM_ITEM_CODE=A.item_code,
 BEGIN_DIM_TIME_PERIOD_CODE = A.begin_period, 
 END_DIM_TIME_PERIOD_CODE = A.end_period
--SELECT DISTINCT SERIES_ID
FROM ap_series A 
WHERE NOT EXISTS 
(SELECT *
 FROM DIM_SERIES D 
 WHERE A.series_id =D.DIM_SERIES_CODE)


END
