UPDATE A 
SET A.PLACE_CENSUS_GEO_ID=P.CENSUS_GEO_ID
FROM STAGE_HEALTH A 
      INNER JOIN DIM_GEOGRAPHY_PLACE P
	  ON CASE A.STATE
	   WHEN 'KY' THEN 'KT' 
	  ELSE A.STATE
	  END =P.STATE_CODE 
	  AND 
	     (A.PLACE_NAME LIKE '%' + P.PLACE_NAME + '%' OR
		  P.PLACE_NAME LIKE '%' + A.PLACE_NAME + '%')
WHERE A.PLACE_CENSUS_GEO_ID IS NULL AND
      A.PLACE_NAME IS NOT NULL
