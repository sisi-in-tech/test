在做insert table 的时候
INNER JOIN AND LEFT OUTER JOIN 同理
需要JOIN两个table 相同值的所有columns ，DO NOT only join 一个column 

TRUNCATE TABLE FACT_CU_INDEX 
ALTER PROCEDURE P_FACT_CU_INDEX 
AS 
BEGIN 

INSERT FACT_CU_INDEX 
(DIM_CU_SERIES_CODE, YR,MTH_PERIOD,VALUE,
DIM_GEOGRAPHIC_AREA_CODE,DIM_CU_ITEM_CODE,YN_SEASONAL,DIM_TIME_ID,
FILE_SOURCE)
SELECT 
DIM_CU_SERIES_CODE=L.SERIES_ID,
YR=L.YEAR,
MTH_PERIOD=L.PERIOD,
VALUE=L.[ value],
DIM_GEOGRAPHIC_AREA_CODE=D.DIM_GEOGRAPHIC_AREA_CODE,
DIM_CU_ITEM_CODE=D.DIM_CU_ITEM_CODE,
YN_SEASONAL=D.SEASONAL,
DIM_TIME_ID=T.DIM_TIME_ID,
FILE_SOURCE= 'LOAD_CU_DATA_CURRENT'

FROM LOAD_CU_DATA_CURRENT L INNER JOIN
	 DIM_CU_SERIES D  
	   ON D.DIM_CU_SERIES_CODE=L.[series_id        ] INNER JOIN 
	 DIM_TIME T 
	 ON T.YR=L.year AND
	    T.DIM_TIME_PERIOD_CODE = L.PERIOD
WHERE NOT EXISTS 
 (SELECT * 
  FROM FACT_CU_INDEX F
  WHERE F.DIM_CU_SERIES_CODE = L.[series_id        ] AND
        F.DIM_TIME_ID = T.DIM_TIME_ID)

END
