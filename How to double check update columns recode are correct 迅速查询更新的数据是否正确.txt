从一个表格导入一行新增的数据到另外一个表格 ，找到相同的column， 在update 列之外
UPDATE A 
SET A.ZIP_CODE_CENSUS_GEO_ID=Z.CENSUS_GEO_ID
FROM STAGE_HEALTH A 
     INNER JOIN DIM_GEOGRAPHY_ZIP_CODE Z 
	 ON A.ZCTA=Z.ZIP_CODE



SELECT SUM(CASE WHEN COUNTY_NAME IS NOT NULL THEN 1 END),
SUM(CASE WHEN FIPS_COUNTY_CD IS NOT NULL THEN 1 END),
 SUM(CASE WHEN COUNTY_CENSUS_GEO_ID IS NOT NULL THEN 1 END)
FROM STAGE_HEALTH
